Implement forEach, map, filter, reduceRight and some using only reduce.

forEach
(fn, list) => reduce((l, value) => {fn(value); return l;}, list, list);
map
(fn, list) => reduce((a, value) => a.concat(fn(value)), [], list);
filter
(fn, list) => reduce((a, value) => a.concat(fn(value)), [], list);
reduceRight
(array, ...args) => [].reduce.apply(array.reverse(), args);
some
(array, ...args) => [].reduce.call(Object.keys(array), args);

smallest number

Make a function to find the smallest number in an array. Use only less function and one of the listed below functions:map, filter, reduce.

function less(a, b) {
  return a < b ? a : b;
}

const smallest = (list) => {
  if (list.length) {
    return reduce(less, list[0], list)
  } else {
    return null;
  }
}
